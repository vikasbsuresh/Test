let
    usage = Text.Lower([UsageType]),

    cleaned = Text.Replace(
        Text.Replace(
            Text.Replace(
                Text.Replace(usage, ":", "-"),
                "_", "-"
            ),
            "/", "-"
        ),
        ".", "-"
    ),

    parts = List.Select(Text.Split(cleaned, "-"), each _ <> ""),

    fullRegionCandidates =
        if List.Count(parts) >= 3 then
            List.Transform(
                {0..List.Count(parts)-3},
                each Text.Combine(List.Range(parts, _, 3), "-")
            )
        else
            {},

    knownFullRegions =
        List.Transform(
            region_map[region_code],
            each Text.Lower(_)
        ),

    shortCodes =
        List.Transform(
            region_map[region_short_code],
            each Text.Lower(_)
        ),

    fullRegionMatch =
        List.First(
            List.Select(
                fullRegionCandidates,
                each List.Contains(knownFullRegions, _)
            ),
            null
        ),

    shortCodeMatch =
        if fullRegionMatch = null then
            List.First(
                List.Select(
                    parts,
                    each List.Contains(shortCodes, _)
                ),
                null
            )
        else
            null,

    regionKey =
        if fullRegionMatch <> null then
            fullRegionMatch
        else if shortCodeMatch <> null then
            shortCodeMatch
        else
            null
in
    regionKey
